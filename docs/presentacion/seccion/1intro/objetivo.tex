	\begin{itemize}
		\only<1>{\item Objetivo General
				\begin{itemize}
					\item Realizar una comunicación entre un FPGA y una PC mediante USB 2.0.
				\end{itemize}}
		\only<2>{\item Objetivos Particulares
				\begin{itemize}
					\item Seleccionar el dispositivo para realizar la interfaz USB.
					\item Realizar la configuración de la interfaz USB.
					\item Sintetizar un circuito en VHDL que sea capaz de interactuar con la interfaz USB, minimizando la utilización de recursos programables.
					\item Desarrollar un sistema de pruebas.
					\item Validar el funcionamiento.
				\end{itemize}}
	\end{itemize}
